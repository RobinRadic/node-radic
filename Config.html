Config

Constructor

 new Config(name, options)

Configuration

Parameters:
Name Type Attributes Description
name string

Name for this config
options object <optional>

Configuration options
Example
{% highlight javascript %}var radic = require('radic'), config = new radic.Config('my-syper-app', { // optional configuration path: '/path/to/config/folder' // defaults to ~/.radic/config }); config.set('c', { b: 'bar' }); config.get('c.b'); //> returns bar config.get('c'); //> returns { b: 'bar' } config.save(); //> configuration will be permanently saved to the config file. config.set('save', 'i will be saved synchronised', true); config.set('save2', 'i will be saved asynchronised', function(){ this.config.set('save3', 'and synchronised again', true); });{% endhighlight %}
